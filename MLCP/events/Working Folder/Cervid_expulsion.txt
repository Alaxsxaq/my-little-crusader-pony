#Cervids forcibly expelled
province_event = {
	id = cervidexp.100
	title = "CERVIDEXP.100"
	desc = "CERVIDEXP.100.DESC"
	picture = GFX_evt_stone_church
	
	trigger = {
		owner = {
			OR = {
				top_liege = k_earth
				top_liege = k_pegasopolis
				top_liege = k_unicornia
			}
			NOT = { culture_group = deer_group }
			NOT = { culture = crystal_culture }
			culture_group = equestrian
		}
		culture_group = deer_group
		NOT = { has_global_flag = expulsion_over }
	}
	
	mean_time_to_happen = {
		months = 120
		modifier = {
			factor = 0.75
			NOT = { num_of_settlements = 2}
		}
		modifier = {
			factor = 1.5
			num_of_settlements = 3
		}
	}
	
	option = {
		name = "CERIVDEXP.100.OPTA"
		
		culture = ROOT
		religion = ROOT
		
	}
}
