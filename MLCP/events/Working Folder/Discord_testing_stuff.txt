
# Herald of Discord
narrative_event = {
	id=DE.1
	title = "EVTNAME_DE001"
	desc="EVTDESC_DE001"
	major=yes

	picture = "GFX_evt_mongols"
	border = "GFX_event_narrative_frame_war"
	
	trigger = {
		NOT = { has_global_flag = Discord_heralded }
		NOT = { has_global_flag = Discord_exists }
		NOT = { has_global_flag = Discord_is_stoned }
		NOT = { has_global_flag = Discord_defeated }
		OR = {
			AND = {
				year = 1150
				AND = {

					k_earth = {
						holder_scope = {
							NOT = {religion_authority = 0.3}
						}
					}
					k_pegasopolis = {
						holder_scope = {
							NOT = {religion_authority = 0.3}
						}
					}
					k_unicornia = {
						holder_scope = {
							NOT = {religion_authority = 0.3}
						}
					}
				}
			}
			
			k_earth = {
				has_holder = no
			}
			k_pegasopolis = {
				has_holder = no
			}
			k_unicornia = {
				has_holder = no
			}
		}
	}
	
	mean_time_to_happen = {
		months = 6
	}

	immediate = {
		set_global_flag = Discord_heralded
		random_province_lord = {
			culture_group = equestrian
			random_demesne_province = {
				set_province_flag = Disharmony_gathers
			}
		}
	}
	
	option = {
		name = "EVTOPTA60000"
	}
}
